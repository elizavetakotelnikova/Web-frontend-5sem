        class Solution {
            public:
            bool isPalindrome(string s) {
                int beginning = 0;
                int end = s.length() - 1;
                while (beginning <= end) {
                    if (!(isdigit(s[beginning]) || isalpha(s[beginning]))) beginning++;
                    else if (!(isdigit(s[end]) || isalpha(s[end]))) end--;
                    else {
                    if (tolower(s[beginning]) != tolower(s[end])) return false;
                    beginning++;
                    end--;
                }
                return true;
            }
        };
